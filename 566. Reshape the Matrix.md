# 566. Reshape the Matrix
####
```
class Solution:
    def matrixReshape(self, nums, r, c):
        """
        :type nums: List[List[int]]
        :type r: int
        :type c: int
        :rtype: List[List[int]]
        """
        a = []
        for i in nums:
            for item in i:
                a.append(item)
        # if the elements in nums cannot make a r*c matrix, it should return the orginal nums
        if len(a) < r*c:
            return nums
        # generate a list that contains 
        ans = [0] * r
        # generate a index point that associates with list a
        count = 0
        for i in range(r):
            ans[i] = [0] * c
            for ix in range(c):
                count += 1
                ans[i][ix] = a[count-1]
        return ans
